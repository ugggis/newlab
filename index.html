        <tbody id="taskTable">

            <tr>

                <td contenteditable="true">Paraugs</td>

                <td contenteditable="true">Students</td>

                <td>

                    <select>

                        <option value="12:00" selected>12:00</option>

                        <option value="13:00">13:00</option>

                        <option value="14:00">14:00</option>

                        <option value="15:00">15:00</option>

                        <option value="16:00">16:00</option>

                    </select>

                </td>

                <td contenteditable="true">Piezīme</td>

            </tr>

        </tbody>

    </table>

    <button onclick="addRow()">Pievienot ierakstu</button>

    <button onclick="saveTable()">Saglabāt</button>

    <button onclick="clearTable()">Notīrīt</button>


    <script>

        const taskTable = document.getElementById("taskTable");

        const workingHours = ["12:00", "13:00", "14:00", "15:00", "16:00"];


        function addRow(task = "", student = "", date = "12:00", notes = "") {

            const tr = document.createElement("tr");

            tr.innerHTML = `

                <td contenteditable="true">${task}</td>

                <td contenteditable="true">${student}</td>

                <td>

                    <select>

                        ${workingHours.map(time => `<option value="${time}" ${time === date ? "selected" : ""}>${time}</option>`).join("")}

                    </select>

                </td>

                <td contenteditable="true">${notes}</td>

            `;

            taskTable.appendChild(tr);

        }


        function saveTable() {

            const rows = [...taskTable.rows].map(row => ({

                task: row.cells[0].innerText,

                student: row.cells[1].innerText,

                date: row.cells[2].querySelector("select").value,

                notes: row.cells[3].innerText

            }));

            localStorage.setItem("taskData", JSON.stringify(rows));

            alert("Tabula saglabāta lokāli!");

        }


        function clearTable() {

            localStorage.removeItem("taskData");

            taskTable.innerHTML = "";

            addRow("Paraugs", "Students", "12:00", "Piezīme");

        }

    </script>
